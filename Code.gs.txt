  if (file === 'manifest') {

      name: "Behind Schedule",

            e?.parameter?.relationship,
            e?.parameter?.sortBy,

            e?.parameter?.page,

// ========== USER AUTHENTICATION ==========
function authenticateUser(username, password) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName('Users');
  if (!sheet) {
    return { success: false, error: 'User database not found' };
  }
  
  const data = sheet.getDataRange().getDisplayValues();
  if (data.length < 2) {
    return { success: false, error: 'No users found' };
  }
  
  const headers = data[0];
  const userIdCol = headers.indexOf('User ID');
  const usernameCol = headers.indexOf('Username');
  const passwordCol = headers.indexOf('Password');
  const statusCol = headers.indexOf('Status');
  
  if (usernameCol === -1 || passwordCol === -1 || statusCol === -1) {
    return { success: false, error: 'Invalid Users sheet format' };
  }
  
  const normalizedUsername = username.trim().toLowerCase();
  
  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    const sheetUsername = (row[usernameCol] || '').trim().toLowerCase();
    const sheetPassword = (row[passwordCol] || '').trim();
    const sheetStatus = (row[statusCol] || '').trim();
    
    if (sheetUsername === normalizedUsername && sheetPassword === password) {
      if (sheetStatus !== 'Active') {
        return { success: false, error: 'Account is not active' };
      }
      return {
        success: true,
        user: {
          userId: row[userIdCol] || '',
          username: sheetUsername,
          status: sheetStatus
        }
      };
    }
  }
  
  return { success: false, error: 'Invalid username or password' };
}

// ========== HELPER FUNCTIONS ==========
const isValid = (val) => { const n = Number(val); return !isNaN(n) && n > 0.0001; };

function getCategoryCriteria() {
  if (CATEGORY_CRITERIA) return CATEGORY_CRITERIA;

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const dropSheet = ss.getSheetByName('Drop');
  if (!dropSheet) return getDefaultCategories();

  let range;
  try { range = ss.getRangeByName('Category'); } catch (e) { range = null; }

  if (!range) {
    const data = dropSheet.getDataRange().getDisplayValues();
    const headerRow = data.findIndex(row => row.includes('DatefROM') && row.includes('DateTo'));
    if (headerRow === -1) return getDefaultCategories();
    const headers = data[headerRow];
    const minCol = headers.indexOf('DatefROM');
    const maxCol = headers.indexOf('DateTo');
    const actualCol = headers.indexOf('Category_Actual');
    const probableCol = headers.indexOf('Category_Probable');
    if (minCol === -1 || maxCol === -1 || actualCol === -1 || probableCol === -1) return getDefaultCategories();
    const rows = data.slice(headerRow + 1).filter(row => row[minCol] && row[maxCol]);
    const criteria = rows.map(row => ({
      min: parseInt(row[minCol]) || 0,
      max: parseInt(row[maxCol]) || 9999999,
      actual: row[actualCol]?.trim() || 'Pass',
      probable: row[probableCol]?.trim() || 'Pass'
    })).sort((a, b) => a.min - b.min);
    CATEGORY_CRITERIA = criteria;
    return criteria;
  }

  const values = range.getDisplayValues();
  const headers = values[0];
  const minCol = headers.indexOf('DatefROM');
  const maxCol = headers.indexOf('DateTo');
  const actualCol = headers.indexOf('Category_Actual');
  const probableCol = headers.indexOf('Category_Probable');
  if (minCol === -1 || maxCol === -1 || actualCol === -1 || probableCol === -1) return getDefaultCategories();
  const criteria = values.slice(1).filter(row => row[minCol] && row[maxCol]).map(row => ({
    min: parseInt(row[minCol]) || 0,
    max: parseInt(row[maxCol]) || 9999999,
    actual: row[actualCol]?.trim() || 'Pass',
    probable: row[probableCol]?.trim() || 'Pass'
  })).sort((a, b) => a.min - b.min);
  CATEGORY_CRITERIA = criteria;
  return criteria;
}

function getDefaultCategories() {
  return [
    { min: 0, max: 29, actual: 'Pass', probable: 'Pass' },
    { min: 30, max: 89, actual: 'WL', probable: 'PWL' },
    { min: 90, max: 9999999, actual: 'NPA', probable: 'PNPA' }
  ];
}

function getSheetData() {
  if (GLOBAL_DATA) return GLOBAL_DATA;
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName("Data");
  if (!sheet) throw new Error("Sheet 'Data' not found.");
  const range = sheet.getDataRange();
  const rawValues = range.getValues();
  const displayValues = range.getDisplayValues();
  const headers = rawValues[0].map(h => String(h).trim());
  const data = [];
  for (let i = 1; i < rawValues.length; i++) {
    let obj = {};
    headers.forEach((header, j) => {
      if (["OVD_DATE", "NEPALI_OVD_Month", "NEPALI_OVD_Year", "CIF_ID", "ACCOUNT_NO", "CUSTOMER_GROUP", "RENEWAL", "CATEGORY_ACTUAL", "CATEGORY_PROBABLE"].includes(header)) {
        obj[header] = String(displayValues[i][j]).trim();
      } else {
        obj[header] = rawValues[i][j];
      }
    });
    data.push(obj);
  }
  GLOBAL_DATA = data;
  return data;
}

function preloadCache() {
  try {
    getSheetData();
    getCategoryCriteria();
    return { status: 'OK' };
  } catch (e) {
    return { error: e.toString() };
  }
}

function getCustomerList() {
  const data = getSheetData();
  const lookup = new Map();
  data.forEach(row => {
    const cif = row.CIF_ID;
    const name = String(row.CUSTOMER_NAME || '').trim();
    if (cif && name && !lookup.has(cif)) {
      lookup.set(cif, { id: cif, name: name });
    }
  });
  return Array.from(lookup.values());
}

function getCustomerMetrics(cifId) {
  const data = getSheetData();
  const records = data.filter(r => r.CIF_ID === cifId);
  if (records.length === 0) return null;

  const first = records[0];
  let maxOvdDays = 0, maxRevOvdDays = 0;
  let netOverdueForCategory = 0;  // Always includes P_OVD for category calculation
  let netOverdueForDisplay = 0;    // Excludes P_OVD for EXPIRED accounts
  
  records.forEach(r => {
    const ovd = Number(r.OVD_DAYS) || 0;
    const rev = Number(r.REV_OVD_DAYS) || 0;
    if (ovd > maxOvdDays) maxOvdDays = ovd;
    if (rev > maxRevOvdDays) maxRevOvdDays = rev;
    
    const iOvd = Number(r.I_OVD) || 0;
    const pOvd = Number(r.P_OVD) || 0;
    const renewal = String(r.RENEWAL || '').trim().toUpperCase();
    
    // For category calculation: ALWAYS include P_OVD
    netOverdueForCategory += iOvd + pOvd;
    
    // For display calculation: Exclude P_OVD if EXPIRED
    if (renewal === 'EXPIRED') {
      netOverdueForDisplay += iOvd;  // Only add I_OVD, exclude P_OVD
    } else {
      netOverdueForDisplay += iOvd + pOvd;  // Add both
    }
  });
  
  const criteria = getCategoryCriteria();
  let actualCat = "Pass", probableCat = "Pass";
  
  // Use netOverdueForCategory (includes P_OVD always) for category determination
  if (Math.abs(netOverdueForCategory) < 0.0001) {
    actualCat = "Pass";
    probableCat = "Pass";
  } else {
    for (let rule of criteria) {
      if (maxOvdDays >= rule.min && maxOvdDays <= rule.max) actualCat = rule.actual;
      if (maxRevOvdDays >= rule.min && maxRevOvdDays <= rule.max) probableCat = rule.probable;
    }
  }
  
  let metrics = {
    CUSTOMER_NAME: first.CUSTOMER_NAME,
    CIF_ID: first.CIF_ID,
    CUSTOMER_GROUP: (first.CUSTOMER_GROUP || "NO GROUP").trim(),
    CATEGORY_ACTUAL: actualCat,
    CATEGORY_PROBABLE: probableCat,
    RENEWAL: first.RENEWAL || "N/A",
    TOT_OVD_AMT: 0,
    MIN_BAL_NPA: 0,
    MIN_BAL_WL: 0,
    MAX_OVD_DAYS: maxOvdDays,
    OP_BALANCE: 0,
    OUTSTANDING_LOAN: 0,
    TOTAL_OVERDUE_NET: 0,
    NET_OVERDUE: netOverdueForDisplay,  // Use display version for UI
    accounts: []
  };

  const accMap = new Map();
  records.forEach(r => {
    const accNo = r.ACCOUNT_NO;
    if (!accNo) return;
    if (!accMap.has(accNo)) {
      accMap.set(accNo, {
        ACCOUNT_NO: accNo,
        SCHEME: r.SCHEME || "N/A",
        OUTSTANDING_LOAN: 0,
        FULL_RATE: r.FULL_RATE || 0,
        MIN_BAL_NPA: 0,
        MIN_BAL_WL: 0,
        SUM_TOT_OVD: 0,
        SUM_PENAL: 0,
        NET_OVD: 0,
        RENEWAL: String(r.RENEWAL || '').trim().toUpperCase()
      });
    }
    let a = accMap.get(accNo);
    const rowLoan = Number(r.OUTSTANDING_LOAN) || 0;
    if (a.OUTSTANDING_LOAN <= 0.0001 && rowLoan > 0.0001) a.OUTSTANDING_LOAN = rowLoan;
    if (isValid(r.MIN_BAL_NPA)) a.MIN_BAL_NPA = Math.max(a.MIN_BAL_NPA, Number(r.MIN_BAL_NPA));
    if (isValid(r.MIN_BAL_WL)) a.MIN_BAL_WL = Math.max(a.MIN_BAL_WL, Number(r.MIN_BAL_WL));
    
    // For TOT_OVD_AMT: Exclude P_OVD if EXPIRED
    const rowOvdAmt = Number(r.TOT_OVD_AMT) || 0;
    const iOvd = Number(r.I_OVD) || 0;
    const pOvd = Number(r.P_OVD) || 0;
    const renewal = a.RENEWAL;
    
    let adjustedOvdAmt = rowOvdAmt;
    if (renewal === 'EXPIRED') {
      adjustedOvdAmt = rowOvdAmt - pOvd;  // Subtract P_OVD from total
    }
    
    if (isValid(adjustedOvdAmt)) a.SUM_TOT_OVD += adjustedOvdAmt;
    
    const rowPenal = Number(r.PENAL_INT) || 0;
    if (isValid(rowPenal)) a.SUM_PENAL += rowPenal;
  });

  const opRow = records.find(r => isValid(r.OP_BALANCE));
  metrics.OP_BALANCE = opRow ? Number(opRow.OP_BALANCE) : 0;

  metrics.accounts = Array.from(accMap.values()).map(a => {
    a.NET_OVD = a.SUM_TOT_OVD - a.SUM_PENAL;
    metrics.OUTSTANDING_LOAN += a.OUTSTANDING_LOAN;
    metrics.TOTAL_OVERDUE_NET += a.NET_OVD;
    metrics.TOT_OVD_AMT += a.SUM_TOT_OVD;
    metrics.MIN_BAL_NPA += a.MIN_BAL_NPA;
    metrics.MIN_BAL_WL += a.MIN_BAL_WL;
    return a;
  });

  return metrics;
}

function getGroupDetails(groupName) {
  const data = getSheetData();
  const trimmedGroup = String(groupName).trim();
  const groupRecords = data.filter(r => String(r.CUSTOMER_GROUP).trim() === trimmedGroup);
  if (groupRecords.length === 0) return null;

  let summary = { groupName: trimmedGroup, totalLoan: 0, totalOvd: 0, totalNetOvd: 0, minNpa: 0, minWl: 0, unitCount: 0 };
  const cifMap = new Map();
  const criteria = getCategoryCriteria();

  groupRecords.forEach(r => {
    if (!cifMap.has(r.CIF_ID)) {
      cifMap.set(r.CIF_ID, {
        name: r.CUSTOMER_NAME, 
        actual: '',
        probable: '',
        groupName: r.CUSTOMER_GROUP, 
        records: []
      });
    }
    cifMap.get(r.CIF_ID).records.push(r);
  });

  summary.unitCount = cifMap.size;
  const units = [];

  cifMap.forEach((cifData, cifId) => {
    let u_loan = 0, u_ovd = 0, u_penal = 0, u_npa = 0, u_wl = 0;
    let accMap = new Map();
    
    let maxOvdDays = 0;
    let maxRevOvdDays = 0;
    let netOverdueForCategory = 0;  // Always includes P_OVD
    let netOverdueForDisplay = 0;    // Excludes P_OVD for EXPIRED

    cifData.records.forEach(r => {
      const accNo = r.ACCOUNT_NO || "UNKNOWN";
      const renewal = String(r.RENEWAL || '').trim().toUpperCase();
      
      if (!accMap.has(accNo)) {
        accMap.set(accNo, { 
          loan: 0, ovd: 0, penal: 0, npa: 0, wl: 0, 
          renewal: renewal,
          iOvdTotal: 0,
          pOvdTotal: 0
        });
      }
      let a = accMap.get(accNo);
     
      const rLoan = Number(r.OUTSTANDING_LOAN) || 0;
      if (a.loan <= 0.0001 && rLoan > 0.0001) a.loan = rLoan;
      
      // Track I_OVD and P_OVD separately
      const iOvd = Number(r.I_OVD) || 0;
      const pOvd = Number(r.P_OVD) || 0;
      a.iOvdTotal += iOvd;
      a.pOvdTotal += pOvd;
      
      // For display OVD: Exclude P_OVD if EXPIRED
      const totOvdAmt = Number(r.TOT_OVD_AMT) || 0;
      if (renewal === 'EXPIRED') {
        if (isValid(totOvdAmt - pOvd)) a.ovd += (totOvdAmt - pOvd);
      } else {
        if (isValid(totOvdAmt)) a.ovd += totOvdAmt;
      }
      
      if (isValid(r.PENAL_INT)) a.penal += Number(r.PENAL_INT);
      if (isValid(r.MIN_BAL_NPA)) a.npa = Math.max(a.npa, Number(r.MIN_BAL_NPA));
      if (isValid(r.MIN_BAL_WL)) a.wl = Math.max(a.wl, Number(r.MIN_BAL_WL));
      
      const ovd = Number(r.OVD_DAYS) || 0;
      const revOvd = Number(r.REV_OVD_DAYS) || 0;
      if (ovd > maxOvdDays) maxOvdDays = ovd;
      if (revOvd > maxRevOvdDays) maxRevOvdDays = revOvd;
      
      // Category calculation: ALWAYS include P_OVD
      netOverdueForCategory += iOvd + pOvd;
      
      // Display calculation: Exclude P_OVD if EXPIRED
      if (renewal === 'EXPIRED') {
        netOverdueForDisplay += iOvd;
      } else {
        netOverdueForDisplay += iOvd + pOvd;
      }
    });

    accMap.forEach(a => { 
      u_loan += a.loan; 
      u_ovd += a.ovd; 
      u_penal += a.penal; 
      u_npa += a.npa; 
      u_wl += a.wl; 
    });
    const u_netOvd = u_ovd - u_penal;

    summary.totalLoan += u_loan; 
    summary.totalOvd += u_ovd; 
    summary.totalNetOvd += u_netOvd;
    summary.minNpa += u_npa; 
    summary.minWl += u_wl;

    let actualCat = "Pass";
    let probableCat = "Pass";
    
    // Use netOverdueForCategory for category determination
    if (Math.abs(netOverdueForCategory) < 0.0001) {
      actualCat = "Pass";
      probableCat = "Pass";
    } else {
      for (let rule of criteria) {
        if (maxOvdDays >= rule.min && maxOvdDays <= rule.max) actualCat = rule.actual;
        if (maxRevOvdDays >= rule.min && maxRevOvdDays <= rule.max) probableCat = rule.probable;
      }
    }

    units.push({
      name: cifData.name, 
      actual: actualCat, 
      probable: probableCat, 
      groupName: cifData.groupName,
      totalOvd: u_ovd, 
      totalNetOvd: u_netOvd, 
      minNpa: u_npa, 
      minWl: u_wl,
      maxOvdDays: maxOvdDays,
      netOverdue: netOverdueForDisplay  // Display version for UI
    });
  });

  return { summary, units };
}

function getOverdueDetails(cifId) {
  const data = getSheetData();
  const records = data.filter(r => r.CIF_ID === cifId);
  const group = {};
  let grandTotal = { TOT_OVD_AMT: 0, I_OVD: 0, P_OVD: 0, PENAL_INT: 0 };
  
  records.forEach(r => {
    const renewal = String(r.RENEWAL || '').trim().toUpperCase();
    const pOvd = Number(r.P_OVD) || 0;
    const iOvd = Number(r.I_OVD) || 0;
    const totOvdAmt = Number(r.TOT_OVD_AMT) || 0;
    const penal = Number(r.PENAL_INT) || 0;
    
    // For EXPIRED accounts, exclude P_OVD from TOT_OVD_AMT
    let adjustedTotOvd = totOvdAmt;
    let adjustedPOvd = pOvd;
    
    if (renewal === 'EXPIRED') {
      adjustedTotOvd = totOvdAmt - pOvd;
      adjustedPOvd = 0;  // Don't count P_OVD separately either
    }
    
    if (!isValid(adjustedTotOvd)) return;
    
    let d = r.OVD_DATE;
    if (!group[d]) group[d] = { OVD_DATE: d, TOT_OVD_AMT: 0, I_OVD: 0, P_OVD: 0, PENAL_INT: 0 };
    
    group[d].TOT_OVD_AMT += adjustedTotOvd;
    group[d].I_OVD += iOvd;
    group[d].P_OVD += adjustedPOvd;
    group[d].PENAL_INT += penal;
    
    grandTotal.TOT_OVD_AMT += adjustedTotOvd;
    grandTotal.I_OVD += iOvd;
    grandTotal.P_OVD += adjustedPOvd;
    grandTotal.PENAL_INT += penal;
  });
  
  return { rows: Object.values(group), total: grandTotal };
}

function getSlicerOptions(cifId) {
  const data = getSheetData();
  const records = data.filter(r => r.CIF_ID === cifId && isValid(r.TOT_OVD_AMT));
  let yearsSet = new Set();
  let monthsSet = new Set();
  records.forEach(r => {
    if (r.NEPALI_OVD_Year) yearsSet.add(String(r.NEPALI_OVD_Year));
    if (r.NEPALI_OVD_Month) monthsSet.add(String(r.NEPALI_OVD_Month));
  });
  const sortedYears = Array.from(yearsSet).sort().reverse();
  const sortedMonths = Array.from(monthsSet).sort((a, b) => NEPALI_MONTHS.indexOf(a) - NEPALI_MONTHS.indexOf(b));
  return { years: sortedYears, months: sortedMonths };
}

function getSlicerSummary(cifId, yearsArray, months) {
  const data = getSheetData();
  const filtered = data.filter(r => {
    const rYear = String(r.NEPALI_OVD_Year);
    const isYear = yearsArray.includes("ALL") || yearsArray.includes(rYear);
    return r.CIF_ID === cifId && isYear && months.includes(r.NEPALI_OVD_Month) && isValid(r.TOT_OVD_AMT);
  });
  const group = {};
  
  filtered.forEach(r => {
    const renewal = String(r.RENEWAL || '').trim().toUpperCase();
    const pOvd = Number(r.P_OVD) || 0;
    const iOvd = Number(r.I_OVD) || 0;
    const totOvdAmt = Number(r.TOT_OVD_AMT) || 0;
    const penal = Number(r.PENAL_INT) || 0;
    
    // For EXPIRED accounts, exclude P_OVD
    let adjustedTotOvd = totOvdAmt;
    let adjustedPOvd = pOvd;
    
    if (renewal === 'EXPIRED') {
      adjustedTotOvd = totOvdAmt - pOvd;
      adjustedPOvd = 0;
    }
    
    let m = r.NEPALI_OVD_Month; 
    if (!group[m]) group[m] = { NEPALI_OVD_Month: m, TOT_OVD_AMT: 0, I_OVD: 0, P_OVD: 0, PENAL_INT: 0 };
    
    group[m].TOT_OVD_AMT += adjustedTotOvd;
    group[m].I_OVD += iOvd;
    group[m].P_OVD += adjustedPOvd;
    group[m].PENAL_INT += penal;
  });
  
  return Object.values(group).sort((a,b) => NEPALI_MONTHS.indexOf(a.NEPALI_OVD_Month) - NEPALI_MONTHS.indexOf(b.NEPALI_OVD_Month));
}

// ========== LOAD FULL DATA (ONE REQUEST) ==========
function loadFullData() {
  const startTime = new Date().getTime();
  
  const data = getSheetData();
  const criteria = getCategoryCriteria();
  
  const customerMap = new Map();
  const groupMap = new Map();
  
  data.forEach(row => {
    const cif = row.CIF_ID;
    if (!cif) return;
    
    const renewal = String(row.RENEWAL || '').trim().toUpperCase();
    const isExpired = renewal === 'EXPIRED';
    
    if (!customerMap.has(cif)) {
      customerMap.set(cif, {
        cifId: cif,
        name: row.CUSTOMER_NAME || 'N/A',
        group: row.CUSTOMER_GROUP || "NO GROUP",
        renewal: renewal,
        maxOvdDays: 0,
        maxRevOvdDays: 0,
        outstandingLoan: 0,
        netOverdueForCategory: 0,  // Always includes P_OVD
        netOverdueForDisplay: 0,    // Excludes P_OVD if EXPIRED
        minBalNpa: 0,
        minBalWl: 0,
        opBalance: 0,
        accounts: new Map()
      });
    }
    
    const c = customerMap.get(cif);
    const ovdDays = Number(row.OVD_DAYS) || 0;
    const revOvdDays = Number(row.REV_OVD_DAYS) || 0;
    const iOvd = Number(row.I_OVD) || 0;
    const pOvd = Number(row.P_OVD) || 0;
    const outstandingLoan = Number(row.OUTSTANDING_LOAN) || 0;
    const minBalNpa = Number(row.MIN_BAL_NPA) || 0;
    const minBalWl = Number(row.MIN_BAL_WL) || 0;
    const opBalance = Number(row.OP_BALANCE) || 0;
    
    c.maxOvdDays = Math.max(c.maxOvdDays, ovdDays);
    c.maxRevOvdDays = Math.max(c.maxRevOvdDays, revOvdDays);
    c.outstandingLoan += outstandingLoan;
    
    // Category calculation: ALWAYS include P_OVD
    c.netOverdueForCategory += iOvd + pOvd;
    
    // Display calculation: Exclude P_OVD if EXPIRED
    if (isExpired) {
      c.netOverdueForDisplay += iOvd;
    } else {
      c.netOverdueForDisplay += iOvd + pOvd;
    }
    
    c.minBalNpa = Math.max(c.minBalNpa, minBalNpa);
    c.minBalWl = Math.max(c.minBalWl, minBalWl);
    if (opBalance > 0) c.opBalance = opBalance;
    
    const accNo = row.ACCOUNT_NO;
    if (accNo && !c.accounts.has(accNo)) {
      const totOvdAmt = Number(row.TOT_OVD_AMT) || 0;
      const penal = Number(row.PENAL_INT) || 0;
      
      // For account display: Exclude P_OVD if EXPIRED
      let adjustedTotOvd = totOvdAmt;
      if (isExpired) {
        adjustedTotOvd = totOvdAmt - pOvd;
      }
      
      c.accounts.set(accNo, {
        accountNo: accNo,
        scheme: row.SCHEME || "N/A",
        outstandingLoan: outstandingLoan,
        fullRate: row.FULL_RATE || 0,
        minBalNpa: minBalNpa,
        minBalWl: minBalWl,
        totOvdAmt: adjustedTotOvd,
        penalInt: penal,
        renewal: renewal
      });
    }
  });
  
  const customers = [];
  const allMetrics = [];
  const groupSummaries = new Map();
  
  customerMap.forEach((c, cif) => {
    // Calculate categories using netOverdueForCategory (always includes P_OVD)
    let actualCat = "Pass", probableCat = "Pass";
    if (Math.abs(c.netOverdueForCategory) >= 0.0001) {
      for (let rule of criteria) {
        if (c.maxOvdDays >= rule.min && c.maxOvdDays <= rule.max) actualCat = rule.actual;
        if (c.maxRevOvdDays >= rule.min && c.maxRevOvdDays <= rule.max) probableCat = rule.probable;
      }
    }
    
    const accounts = Array.from(c.accounts.values()).map(a => ({
      ACCOUNT_NO: a.accountNo,
      SCHEME: a.scheme,
      OUTSTANDING_LOAN: a.outstandingLoan,
      FULL_RATE: a.fullRate,
      MIN_BAL_NPA: a.minBalNpa,
      MIN_BAL_WL: a.minBalWl,
      TOT_OVD_AMT: a.totOvdAmt,
      PENAL_INT: a.penalInt,
      NET_OVD: a.totOvdAmt - a.penalInt
    }));
    
    const totalOvdAmt = accounts.reduce((sum, a) => sum + a.TOT_OVD_AMT, 0);
    
    const metrics = {
      CIF_ID: cif,
      CUSTOMER_NAME: c.name,
      CUSTOMER_GROUP: c.group,
      RENEWAL: c.renewal,
      CATEGORY_ACTUAL: actualCat,
      CATEGORY_PROBABLE: probableCat,
      MAX_OVD_DAYS: c.maxOvdDays,
      OUTSTANDING_LOAN: c.outstandingLoan,
      TOTAL_OVERDUE_NET: c.netOverdueForDisplay,  // Display version
      TOT_OVD_AMT: totalOvdAmt,
      MIN_BAL_NPA: c.minBalNpa,
      MIN_BAL_WL: c.minBalWl,
      OP_BALANCE: c.opBalance,
      accounts: accounts
    };
    
    allMetrics.push(metrics);
    customers.push({ id: cif, name: c.name });
    
    if (c.group && c.group.toUpperCase() !== "NO GROUP") {
      if (!groupSummaries.has(c.group)) {
        groupSummaries.set(c.group, {
          groupName: c.group,
          unitCount: 0,
          totalLoan: 0,
          totalOvd: 0,
          totalNetOvd: 0,
          minNpa: 0,
          minWl: 0,
          units: []
        });
      }
      const g = groupSummaries.get(c.group);
      g.unitCount++;
      g.totalLoan += c.outstandingLoan;
      g.totalOvd += totalOvdAmt;
      g.totalNetOvd += c.netOverdueForDisplay;  // Display version
      g.minNpa += c.minBalNpa;
      g.minWl += c.minBalWl;
      g.units.push({
        name: c.name,
        actual: actualCat,
        probable: probableCat,
        totalOvd: totalOvdAmt,
        totalNetOvd: c.netOverdueForDisplay,
        minNpa: c.minBalNpa,
        minWl: c.minBalWl,
        maxOvdDays: c.maxOvdDays
      });
    }
  });
  
  const groups = [];
  groupSummaries.forEach(g => {
    groups.push({
      summary: {
        groupName: g.groupName,
        unitCount: g.unitCount,
        totalLoan: g.totalLoan,
        totalOvd: g.totalOvd,
        totalNetOvd: g.totalNetOvd,
        minNpa: g.minNpa,
        minWl: g.minWl
      },
      units: g.units
    });
  });
  
  const executionTime = new Date().getTime() - startTime;
  Logger.log(`Full data loaded in ${executionTime}ms - ${customers.length} customers, ${groups.length} groups`);
  
  return {
    customers: customers,
    allMetrics: allMetrics,
    groups: groups,
    timestamp: new Date().getTime(),
    executionTime: executionTime,
    totalCustomers: customers.length,
    totalGroups: groups.length
  };
}

// ========== CATEGORY TABLE ENDPOINT ==========
function getCustomersByCategory(category, relationship, sortBy, sortOrder, page, pageSize) {
  category = category || 'ALL';
  relationship = relationship || 'ALL';
  sortBy = sortBy || 'name';
  sortOrder = sortOrder || 'asc';
  page = parseInt(page) || 1;
  pageSize = parseInt(pageSize) || 10;

  const data = getSheetData();
  const criteria = getCategoryCriteria();

  const custMap = new Map();

  data.forEach(row => {
    const cif = row.CIF_ID;
    if (!cif) return;

    if (!custMap.has(cif)) {
      custMap.set(cif, {
        cifId: cif,
        name: row.CUSTOMER_NAME || 'N/A',
        relationship: row.CUST_RELATIONSHIP || '',
        maxOvdDays: 0,
        maxRevOvdDays: 0,
        outstandingLoan: 0,
        actualCat: 'Pass',
        probableCat: 'Pass',
        netOverdueForCategory: 0  // Always includes P_OVD
      });
    }

    const cust = custMap.get(cif);
    const ovd = Number(row.OVD_DAYS) || 0;
    const revOvd = Number(row.REV_OVD_DAYS) || 0;
    if (ovd > cust.maxOvdDays) cust.maxOvdDays = ovd;
    if (revOvd > cust.maxRevOvdDays) cust.maxRevOvdDays = revOvd;

    const iOvd = Number(row.I_OVD) || 0;
    const pOvd = Number(row.P_OVD) || 0;
    // Category calculation: ALWAYS include P_OVD
    cust.netOverdueForCategory += iOvd + pOvd;

    cust.outstandingLoan += Number(row.OUTSTANDING_LOAN) || 0;
  });

  const customers = [];
  custMap.forEach(cust => {
    // Use netOverdueForCategory for category determination
    if (Math.abs(cust.netOverdueForCategory) < 0.0001) {
      cust.actualCat = 'Pass';
      cust.probableCat = 'Pass';
    } else {
      for (let rule of criteria) {
        if (cust.maxOvdDays >= rule.min && cust.maxOvdDays <= rule.max) {
          cust.actualCat = rule.actual;
          break;
        }
      }
      for (let rule of criteria) {
        if (cust.maxRevOvdDays >= rule.min && cust.maxRevOvdDays <= rule.max) {
          cust.probableCat = rule.probable;
          break;
        }
      }
    }

    if (relationship !== 'ALL') {
      const custRel = (cust.relationship || '').trim().toUpperCase();
      if (custRel !== relationship.toUpperCase()) return;
    }

    let include = false;
    switch (category.toUpperCase()) {
      case 'ALL': include = true; break;
      case 'NPA': include = cust.actualCat.toUpperCase() === 'NPA'; break;
      case 'WL': include = cust.actualCat.toUpperCase() === 'WL'; break;
      case 'PWL': include = cust.probableCat.toUpperCase() === 'PWL'; break;
      case 'PNPA': include = cust.probableCat.toUpperCase() === 'PNPA'; break;
      case 'PASS': include = cust.actualCat.toUpperCase() === 'PASS'; break;
      default: include = true;
    }
    if (!include) return;

    customers.push({
      cifId: cust.cifId,
      name: cust.name,
      limit: cust.outstandingLoan,
      relationship: cust.relationship,
      actualCat: cust.actualCat,
      probableCat: cust.probableCat
    });
  });

  customers.sort((a, b) => {
    let valA, valB;
    if (sortBy === 'name') {
      valA = a.name.toUpperCase();
      valB = b.name.toUpperCase();
    } else {
      valA = a.limit;
      valB = b.limit;
    }
    if (valA < valB) return sortOrder === 'asc' ? -1 : 1;
    if (valA > valB) return sortOrder === 'asc' ? 1 : -1;
    return 0;
  });

  const total = customers.length;
  const start = (page - 1) * pageSize;
  const end = Math.min(start + pageSize, total);
  const paginated = customers.slice(start, end);

  return {
    customers: paginated,
    total: total,
    page: page,
    pageSize: pageSize,
    totalPages: Math.ceil(total / pageSize)
  };
}

function saveGpsRecord(params) {
  if (!params) return { success: false, error: 'Missing parameters' };
  
  const { cifId, locationType, label, latitude, longitude, customerName } = params;
  
  // For Residence and Business, only one record allowed → overwrite the first existing one
  if (locationType === 'Residence' || locationType === 'Business') {
    const existing = getGpsRecords(cifId).records.filter(r => r.LocationType === locationType);
    if (existing.length > 0) {
      return updateGpsRecord({ 
        rowIndex: existing[0].rowIndex,
        cifId, locationType, label, latitude, longitude, customerName 
      });
    }
  }
  
  const sheet = getGpsSheet_();
  sheet.appendRow([
    customerName || '',
    cifId,
    locationType,
    label || (locationType === 'Business' ? 'Business' : ''),
    parseFloat(latitude) || 0,
    parseFloat(longitude) || 0,
    new Date()
  ]);
  return { success: true };
}

function saveCommitment(params) {
  if (!params) return { success: false, error: 'Missing parameters' };
  const cifId = params.cifId;
  const customerName = params.customerName;
  const date = params.date;
  const remarks = params.remarks;
  let rowIndex = parseInt(params.rowIndex) || -1;
  
  const sheet = getCommitmentsSheet_();
  const data = sheet.getDataRange().getValues();
  const headers = data[0];
  const cifCol = headers.indexOf('CIF_ID');
  const dateCol = headers.indexOf('CommitmentDate');
  
  const targetDateStr = String(date).trim();
  
  if (rowIndex >= 2) {
    sheet.getRange(rowIndex, headers.indexOf('CommitmentDate') + 1).setValue(date);
    sheet.getRange(rowIndex, headers.indexOf('Remarks') + 1).setValue(remarks || '');
    sheet.getRange(rowIndex, headers.indexOf('TimeStamp') + 1).setValue(new Date());
    return { success: true };
  }
  
  for (let i = 1; i < data.length; i++) {
    const rowDateVal = data[i][dateCol];
    if (!rowDateVal) continue;
    let rowDateStr = "";
    try {
      const rowDate = new Date(rowDateVal);
      rowDateStr = Utilities.formatDate(rowDate, Session.getScriptTimeZone(), "yyyy-MM-dd");
    } catch(e) {
      rowDateStr = String(rowDateVal).trim();
    }
    if (String(data[i][cifCol]).trim() === String(cifId).trim() && rowDateStr === targetDateStr) {
      rowIndex = i + 1;
      break;
    }
  }
  
  if (rowIndex !== -1) {
    sheet.getRange(rowIndex, headers.indexOf('Remarks') + 1).setValue(remarks || '');
    sheet.getRange(rowIndex, headers.indexOf('TimeStamp') + 1).setValue(new Date());
  } else {
    sheet.appendRow([
      cifId,
      customerName || '',
      date,
      remarks || '',
      new Date()
    ]);
  }
  return { success: true };
}

// ========== IMAGE CAPTURE & GITHUB UPLOAD ==========
function uploadImage(params) {
  if (!params || !params.cifId || !params.imageData) {
    return { success: false, error: 'Missing parameters' };
  }

  const props = PropertiesService.getScriptProperties();
  let pat = props.getProperty('GITHUB_PAT');
  let owner = props.getProperty('GITHUB_OWNER');
  let repo = props.getProperty('GITHUB_REPO');

  if (!pat) return { success: false, error: 'GitHub PAT not configured in script properties' };
  if (!owner) owner = 'user';
  if (!repo) repo = 'bs-main';

  const folder = 'crm-images';
  const ext = 'webp';
  const fileName = params.fileName || `${params.cifId}_${new Date().getTime()}.${ext}`;

  const url = `https://api.github.com/repos/${encodeURIComponent(owner)}/${encodeURIComponent(repo)}/contents/${encodeURIComponent(folder)}/${encodeURIComponent(fileName)}`;

  const options = {
    method: 'put',
    headers: {
      'Authorization': 'Bearer ' + pat,
      'Content-Type': 'application/json'
    },
    payload: JSON.stringify({
      message: `Upload CRM image ${fileName}`,
      content: params.imageData,
      branch: 'main'
    }),
    muteHttpExceptions: true
  };

  try {
    const response = UrlFetchApp.fetch(url, options);
    const code = response.getResponseCode();
    const result = JSON.parse(response.getContentText());

    if (code < 200 || code >= 300) {
      return { success: false, error: result.message || 'GitHub API error' };
    }

    const downloadUrl = result.content.download_url;

    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let sheet = ss.getSheetByName('Images');
    if (!sheet) {
      sheet = ss.insertSheet('Images');
      sheet.getRange(1, 1, 1, 4).setValues([['CIF_ID', 'FileName', 'URL', 'TimeStamp']]);
      sheet.getRange(1, 1, 1, 4).setFontWeight('bold');
    }
    sheet.appendRow([params.cifId, fileName, downloadUrl, new Date()]);

    return { success: true, url: downloadUrl, fileName: fileName };
  } catch (e) {
    return { success: false, error: e.toString() };
  }
}

function getImages(cifId) {
  if (!cifId) return { images: [] };
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName('Images');
  if (!sheet) return { images: [] };

  const data = sheet.getDataRange().getValues();
  if (data.length < 2) return { images: [] };

  const headers = data[0];
  const cifCol = headers.indexOf('CIF_ID');
  const fileCol = headers.indexOf('FileName');
  const urlCol = headers.indexOf('URL');
  const timeCol = headers.indexOf('TimeStamp');

  if (cifCol === -1 || urlCol === -1) return { images: [] };

  const images = [];
  for (let i = 1; i < data.length; i++) {
    if (String(data[i][cifCol]).trim() === String(cifId).trim()) {
      images.push({
        fileName: data[i][fileCol] || '',
        url: data[i][urlCol] || '',
        timestamp: data[i][timeCol] ? String(data[i][timeCol]) : ''
      });
    }
  }
  return { images: images };
}

function deleteImage(fileName) {
  if (!fileName) return { success: false, error: 'Missing file name' };

  const props = PropertiesService.getScriptProperties();
  let pat = props.getProperty('GITHUB_PAT');
  let owner = props.getProperty('GITHUB_OWNER');
  let repo = props.getProperty('GITHUB_REPO');

  if (!pat) return { success: false, error: 'GitHub PAT not configured' };
  if (!owner) owner = 'user';
  if (!repo) repo = 'bs-main';

  const folder = 'crm-images';

  try {
    const getUrl = `https://api.github.com/repos/${encodeURIComponent(owner)}/${encodeURIComponent(repo)}/contents/${encodeURIComponent(folder)}/${encodeURIComponent(fileName)}`;
    const getRes = UrlFetchApp.fetch(getUrl, {
      headers: { 'Authorization': 'Bearer ' + pat },
      muteHttpExceptions: true
    });

    if (getRes.getResponseCode() >= 400) {
      return { success: false, error: 'File not found on GitHub' };
    }

    const fileInfo = JSON.parse(getRes.getContentText());

    const delUrl = `https://api.github.com/repos/${encodeURIComponent(owner)}/${encodeURIComponent(repo)}/contents/${encodeURIComponent(folder)}/${encodeURIComponent(fileName)}`;
    const delOptions = {
      method: 'delete',
      headers: {
        'Authorization': 'Bearer ' + pat,
        'Content-Type': 'application/json'
      },
      payload: JSON.stringify({
        message: 'Delete CRM image ' + fileName,
        sha: fileInfo.sha,
        branch: 'main'
      }),
      muteHttpExceptions: true
    };

    UrlFetchApp.fetch(delUrl, delOptions);

    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = ss.getSheetByName('Images');
    if (sheet) {
      const data = sheet.getDataRange().getValues();
      for (let i = data.length - 1; i >= 1; i--) {
        if (String(data[i][1]).trim() === String(fileName).trim()) {
          sheet.deleteRow(i + 1);
          break;
        }
      }
    }

    return { success: true };
  } catch (e) {
    return { success: false, error: e.toString() };
  }
}
